# código simple para comprobar su correcto funcionamiento
Archivo > Ejemplos > Basics > Blink                          # en el caso de que no funcione lanzo el código de abajo 
+----------------------------------------------------------+
// Definimos el pin del LED interno
const int ledPin = 2; // El LED interno del ESP32 está en GPIO 2
void setup() {
  // Inicializamos el pin del LED como salida
  pinMode(ledPin, OUTPUT); 
  // Encendemos el LED al inicio
  digitalWrite(ledPin, HIGH);
  // Inicializamos la comunicación serie a 115200 baudios
  Serial.begin(115200);
}
void loop() {
  // Imprimimos "hola" en el monitor serie
  Serial.println("hola");
  
  // Esperamos 500 milisegundos
  delay(500);
}
+----------------------------------------------------------+
