 ==CONFIGURAR UNA NUEVA CUOTA==

## primero creo un proyecto
# asocio un ID del proyecto con el directorio existente
echo "002:/datos/SalaDeReuniones" >> /etc/projects

# asocia el nombre del proyecto con el ID
echo "SalaDeReuniones:002" >> /etc/projid

## activar el proyecto
xfs_quota -x -c 'project -s SalaDeReuniones' /datos

## establecer los límites de cuota
xfs_quota -x -c 'limit -p bsoft=15g bhard=15g SalaDeReuniones' /datos

## Verificar la configuración
xfs_quota -xc 'report -pbih'

--------------------------------------------------------------------------------

 ==AMPLIAR UNA CUOTA EXISTENTE==

## OPCIONAL: comprobar el espacio de /datos
df -T /datos

## buscar el ID asociado al proyecto existente (los proyectos tienen el mismo nombre que los directorios)
cat /etc/projid

## cambiamos el espacio del proyecto, en este caso de Logistica:016
# bsoft: espacio máximo con tiempo de gracia limitado
# bhard: espacio máximo del espacio en disco que no se puede superar
xfs_quota -x -c 'limit -p bsoft=20g bhard=23g 016' /datos

# reiniciar
